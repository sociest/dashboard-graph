import requests
from bs4 import BeautifulSoup
import logging
def get_search_content(page):
    url = f"https://sisin.oep.org.bo/?search_title=&search_type=all&search_media=all&search_category=all&search_date1=&search_date2=&page={page}"
    response = requests.get(url, timeout=60, headers={"User-Agent": "Mozilla/5.0"})
    soup = BeautifulSoup(response.text, "html.parser")
    return soup
def extract_search_information(soup):
    news_posts = []

    posts = soup.find_all("div", class_="list-news")

    for post in posts:
        # Título
        title_tag = post.find("h2", class_="entry-title")
        title = title_tag.a.text.strip() if title_tag and title_tag.a else None

        # Enlace
        link = title_tag.a["href"] if title_tag and title_tag.a else None

        # Medio
        media_tag = post.find("li", class_="namemedia-badge")
        media = media_tag.a.text.strip() if media_tag and media_tag.a else None

        # Fecha
        date_tag = post.find_all("li")[1]
        date = date_tag.a.text.strip() if date_tag and date_tag.a else None

        # Categoría
        category_tag = post.find_all("li")
        category = (
            category_tag[2].a.text.strip()
            if len(category_tag) > 2 and category_tag[2].a
            else None
        )

        # Resumen
        summary_tag = post.find("p")
        summary = summary_tag.text.strip() if summary_tag else None

        # Imágen
        author_div = post.find("div", class_="author")
        img_tag = author_div.find("img") if author_div else None
        image_url = img_tag["src"] if img_tag else None

        # Imprimir o guardar la información
        news_posts.append(
            {
                "title": title,
                "link": link,
                "media": media,
                "date": date,
                "category": category,
                "summary": summary,
                "image_url": image_url,
            }
        )
    return news_posts
def get_total_pages(soup):
    pagination = soup.find("ul", class_="pagination")
    if pagination:
        page_links = pagination.find_all("a")
        if page_links:
            # El antepenúltimo <li> con <a> suele tener el número de la última página
            last_page_link = page_links[-2]
            return int(last_page_link.text.strip())
    return 1  # Si no hay paginación, asumimos que solo hay una página
def get_search_results(limit=None):
    all_news_posts = []
    soup = get_search_content(1)
    pages = get_total_pages(soup)
    news_posts = extract_search_information(soup)
    all_news_posts.extend(news_posts)
    
    limit = limit if limit is not None else pages
    for page in range(2, limit + 1):
        print(page,limit,page/limit)
        soup = get_search_content(page)
        news_posts = extract_search_information(soup)
        all_news_posts.extend(news_posts)
    
    return all_news_posts
def get_detail_content(url):
    response = requests.get(url, timeout=60, headers={"User-Agent": "Mozilla/5.0"})
    soup = BeautifulSoup(response.text, "html.parser")
    if "medios_redes_sociales" in url:
        content_type = "redes_sociales"
    elif "medios_escritos" in url:
        content_type = "medios_escritos"
    elif "medios_radiales" in url:
        content_type = "medios_radiales"
    elif "medios_televisivos" in url:
        content_type = "medios_televisivos"
    elif "medios_digitales" in url:
        content_type = "medios_digitales"
    else:
        content_type = "desconocido"

    return soup, content_type
def extract_detail_information_redes_sociales(soup):
    media_tag = soup.find("li", class_="namemedia-badge")

    # Obtener la fecha de publicación usando el segundo <li> en el <ul>
    entry_meta = soup.select_one("div.entry-meta ul")
    date = None
    if entry_meta:
        li_tags = entry_meta.find_all("li")
        if len(li_tags) > 1 and li_tags[1].a:
            date = li_tags[1].a.text.strip()

    title = soup.find("h1", class_="entry-title").text.strip() if soup.find("h1", class_="entry-title") else None

    # Extraer categoría y subcategoría si existen
    categoria = None
    subcategoria = None
    entry_meta = soup.select_one("div.entry-meta ul")
    if entry_meta:
        li_tags = entry_meta.find_all("li")
        if len(li_tags) > 2 and li_tags[2].a:
            categoria = li_tags[2].a.text.strip()
        if len(li_tags) > 3 and li_tags[3].a:
            subcategoria = li_tags[3].a.text.strip()

    # Extraer url de redes sociales
    url = None
    visit_link = soup.find("a", onclick=True)
    if visit_link and "window.open" in visit_link["onclick"]:
        import re
        match = re.search(r"window\.open\('([^']+)'", visit_link["onclick"])
        if match:
            url = match.group(1)
    
    return {
        "media": media_tag.a.text.strip() if media_tag and media_tag.a else None,
        "date": date,
        "type": type,
        "title": title,
        "categoria": categoria,
        "subcategoria": subcategoria,
        "url": url
    }
